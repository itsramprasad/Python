# Implement derived classes like Dog, Cat, and Bird that override the make_sound() 
# method to produce different sounds. Demonstrate polymorphism by calling 
# the method on objeccts of different classes.

# Code: 

# Program to demonstrate polymorphism in Python
class Animal:
    def make_sound(self):
        raise NotImplementedError("Subclass must implement abstract method 'make_sound'")
class Dog(Animal):
    def make_sound(self):
        return "Bow! Bow!"
class Cat(Animal):
    def make_sound(self):
        return "Mew! Mew!"
class Bird(Animal):
    def make_sound(self):
        return "Tweet! Tweet!"
# Demonstrating polymorphism
print("\n--- Polymorphism Demonstration ---")
animals = [Dog(), Cat(), Bird()]
for animal in animals:
    print(f"An animal makes sound: {animal.make_sound()}")

# Output: --- Polymorphism Demonstration ---
#         An animal makes sound: Bow! Bow!
#         An animal makes sound: Mew! Mew!
#         An animal makes sound: Tweet! Tweet!
